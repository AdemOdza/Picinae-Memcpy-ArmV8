Definition time_of_addr (s : store) (a : addr) : N :=
	match a with
		| 0x00101140	=>movzx_r32_r16
		| 0x00101143	=>test_r16_r16
		| 0x00101146	=>jz_addr
		| 0x00101148	=>xor_r32_r32
		| 0x0010114a	=>jmp_addr
		| 0x00101150	=>add_r64_0x1
		| 0x00101154	=>cmp_r32_r32
		| 0x00101156	=>jle_addr
		| 0x00101158	=>cmp_[m32]_r32
		| 0x0010115b	=>jnz_addr
		| 0x0010115d	=>ret
		| 0x00101160	=>mov_r32_i
		| 0x00101165	=>ret
