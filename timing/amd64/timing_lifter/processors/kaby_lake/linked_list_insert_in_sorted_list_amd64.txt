Definition time_of_addr (s : store) (a : addr) : N :=
	match a with
		| 0x00101340	=>mov_wc
		| 0x00101342	=>cmp_r32_i
		| 0x00101347	=>jz_addr
		| 0x00101349	=>nop_[m32]
		| 0x00101350	=>mov_r64_r64
		| 0x00101353	=>mov_wc
		| 0x00101357	=>cmp_r32_i
		| 0x00101359	=>jnc_addr
		| 0x0010135b	=>mov_wc
		| 0x0010135f	=>mov_wc
		| 0x00101363	=>ret
		| 0x00101368	=>mov_r64_r64
		| 0x0010136b	=>mov_wc
		| 0x0010136f	=>mov_wc
		| 0x00101373	=>mov_wc
		| 0x00101377	=>ret
